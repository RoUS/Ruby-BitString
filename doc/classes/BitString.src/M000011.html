  def resize(bits)
    unless (bits.kind_of?(Integer))
      _raise(BitsRInts)
    end
    unless (bits > 0)
      _raise(NoDeficitBits)
    end

    value = @value
    length = self.length
    bs = self.class.new(value, length)
    diffbits = bits - length
    diffbits < 0 ? bs.shrink!(-diffbits) : bs.grow!(diffbits)
  end
